#!/bin/bash
#### Read the array values with space


# 1.
## put string array into $value
value=$(ls -1v);

## get length of $value array
len=${#value[@]};

## Use for loop
for (( i=0; i<$len; i++ )); do echo "${value[$i]}" ; done


# 2.
#### Oder als Einzeiler:
for val in "$(ls -1v)"; do echo -e "$val \n"; done
